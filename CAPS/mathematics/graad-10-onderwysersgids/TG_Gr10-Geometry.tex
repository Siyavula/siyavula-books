 \begin{solutions}{}{
\begin{enumerate}[itemsep=5pt, label=\textbf{\arabic*}. ] 


\item \begin{multicols}{2} %solution 1
    $a=138^{\circ}$\\ $b=42^{\circ}$\\ $c=138^{\circ}$\\ $d=138^{\circ}$\\ $e=42^{\circ}$\\ $f=138^{\circ}$\\ $g=42^{\circ}$
\end{multicols}
\item \begin{multicols}{2}%solution 2
 $\hat{B}_1=110^{\circ}$\\ $\hat{C}_1=80^{\circ}$\\ $\hat{C}_2=30^{\circ}$\\ $\hat{C}_3=70^{\circ}$\\ $\hat{D}_1=100^{\circ}$\\ $\hat{F}_1=70^{\circ}$\\ $\hat{F}_2=30^{\circ}$\\ $\hat{F}_3=80^{\circ}$\\ $\hat{G}_1=70^{\circ}$\\ $\hat{G}_2=30^{\circ}$\\ $\hat{G}_3=80^{\circ}$
\end{multicols}
\item %solution 3
$x + 60^\circ + x - 20^\circ = 180^\circ\\
2x = 180^\circ - 40^\circ\\
2x = 140^\circ\\
\therefore x = 70^\circ$

\item %solution 4
\begin{enumerate}[itemsep=3pt, label=\textbf{(\alph*)} ]
\item Vir binnehoeke $115^\circ + 55^\circ = 170^\circ$\\
as $\parallel$ die som sou $180^\circ$ gewees het $ \therefore$ is die lyne nie ewewydig nie.

\item $K_2 = 180^{\circ} - 124^\circ  - 56^\circ$\\
as $MN \parallel OP$ dan sal $\hat{K}_2 = 56$ gelyk an $\hat{L} = 45^\circ$ wees\\
$\therefore$ is $ MN$ en $OP$ nie ewewydig nie.

\item Laat $\hat{U}$  die kruispunt van lyne $KL$ en $TY$ wees en laat $\hat{V}$ die kruispunt van lyne $KL$ en $MN$ wees.\\
$\hat{U_4} = 96^{\circ}\\
\therefore \hat{U_1} = 180^{\circ} - 95^{\circ}$~~~~~~~(Hoeke op 'n reguitlyn)\\
$\therefore \hat{U_1} = 85^{\circ}$\\
Also, $\hat{U_1} = 85^{\circ} = \hat{V_4}$\\
Hierdie is ooreenkomstige hoeke\\
$\therefore TY \parallel MN$
\end{enumerate}

\item %solution 5
As $a = 2$\\
en $b = a$\\
dan weet ons dat $b = 2$\\
Soortgelyk as $AB \parallel CD$\\
en $EF \parallel AB$\\
dan weet ons dat $EF \parallel CD$
\end{enumerate}}
\end{solutions}
